entity type ElectricVehicle mixes Battery schema name "EV" {
    // {@link Battery} inside the EV
    battery: !Battery

    // Trip Energy required (in kWh)
    tripEnergyRequired: !float
    
    // Risk Aversion (0: No RA, 0.001: Intermediate, 0.005: Risk Averse User)
    K: !float

    // Wrap Around Cost
    wrapAroundCost: !float

    // Length of trip
    tripDuration: int

    // Trip Start (time slot)
    tripStart: int

    // Trip End
    tripEnd: int
}
